const { default: makeWASocket, useMultiFileAuthState, DisconnectReason } = require('@whiskeysockets/baileys'); const { OpenAI } = require('openai'); const P = require('pino'); const fs = require('fs'); const path = require('path'); // CONFIGURAÇÃO DA OPENAI (NÃO COMPARTILHE ESTA CHAVE NO HTML) const openai = new OpenAI({ apiKey: "SUA_CHAVE_DA_OPENAI_AQUI" // <-- Você coloca sua chave aqui no servidor }); // PERSONA DO ROBÔ (Técnicas de Vendas e Conversão) const systemPrompt = ` Você é a Assistente Virtual da 'Estética Pro', empresa que cria sites profissionais e sistemas de agendamento para Esteticistas e Clínicas de Estética. Seu objetivo é converter o lead em cliente (fechar a venda do plano de R$ 130/mês). Seja sempre educada, empática e use técnicas de vendas SPIN ou Gatilhos Mentais. Destaque que o plano não tem fidelidade, inclui hospedagem, domínio, painel de agendamento, controle de estoque e suporte 24h. Se o cliente hesitar, ofereça tirar dúvidas. Se ele fechar, peça o número para contato humano ou direcione para o link: https://wa.me/5511957130312. Responda sempre em texto, de forma curta e direta. `; async function connectToWhatsApp() { const { state, saveCreds } = await useMultiFileAuthState('auth_info_baileys'); const sock = makeWASocket({ auth: state, logger: P({ level: 'silent' }), }); sock.ev.on('creds.update', saveCreds); sock.ev.on('connection.update', (update) => { const { connection, lastDisconnect, qr } = update; if(qr) { console.log('Escaneie este QR Code no WhatsApp Business:'); require('qrcode-terminal').generate(qr, {small: true}); } if(connection === 'close') { const shouldReconnect = (lastDisconnect?.error)?.output?.statusCode !== DisconnectReason.loggedOut; if(shouldReconnect) connectToWhatsApp(); } }); sock.ev.on('messages.upsert', async ({ messages }) => { const msg = messages[0]; if (!msg.message || msg.key.fromMe) return; const sender = msg.key.remoteJid; let userText = ""; // LÓGICA PARA ECONOMIZAR CRÉDITOS: SE FOR ÁUDIO, TRANSCREVE if (msg.message.audioMessage) { try { // Baixar o áudio const buffer = await sock.downloadMediaMessage(msg); const audioPath = path.resolve(__dirname, 'audio.ogg'); fs.writeFileSync(audioPath, buffer); // Enviar para OpenAI Whisper (Transcrição - Super barato) const transcription = await openai.audio.transcriptions.create({ file: fs.createReadStream(audioPath), model: "whisper-1", }); userText = transcription.text; console.log("Cliente enviou áudio transcrito:", userText); } catch (err) { console.error("Erro ao processar áudio:", err); await sock.sendMessage(sender, { text: "Desculpe, não consegui escutar seu áudio. Pode digitar?" }); return; } } else if (msg.message.conversation) { userText = msg.message.conversation; } if (!userText) return; // CHAMADA PARA O GPT-4o-mini (Modelo barato e inteligente) try { const completion = await openai.chat.completions.create({ model: "gpt-4o-mini", // Mais barato que o GPT-4 padrão e excelente para vendas messages: [ { role: "system", content: systemPrompt }, { role: "user", content: userText } ], temperature: 0.7, }); const replyText = completion.choices[0].message.content; // RESPONDE NO WHATSAPP EM TEXTO (Economiza créditos de voz) await sock.sendMessage(sender, { text: replyText }); } catch (err) { console.error("Erro na OpenAI:", err); await sock.sendMessage(sender, { text: "Ops, estou com uma leve instabilidade no sistema agora. Pode aguardar um segundinho?" }); } }); } connectToWhatsApp();
